module main
import sys.object.*        
import sys.console as console    

tellPrime(int n) : string {
    if n ≤ 1 { return 'bad' }           
    for i = 2, i * i ≤ n, i++ {
    if n % i == 0 { return 'bad' }
    }
    return 'good'
}

void main() {
    print = console.writeLine
    print(tellPrime(17))
}
